```py
# Append this code at the bottom of the script, right after the driver.
# You MUST keep the driver code!!
_x = 0
_d = -9

while 1
    clear(1)
    texts(Str(_x), 0, 50, 5, 2, 2)
    Circle(0, _x, 50, 5)
    _x=_x+_d
    if(_x < 0 || _x > 128)
        _d = _d * -1
    end
    show()
wend
```
