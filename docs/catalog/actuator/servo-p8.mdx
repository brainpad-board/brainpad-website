# Append this code at the bottom of the script, right after the driver.
# You MUST keep the driver code!!
dim b1[8] = [0, 45, 90, 135, 180, 135, 90, 45] # degree 
_i = 0

while 1
    for x in range (1, 9)
        servost(x, b1[_i])    
    next
    _i = (_i + 1) % 8
    
    Wait(1000)
    
wend
