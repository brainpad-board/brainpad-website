# Append this code at the bottom of the script, right after the driver.

_i = 1
while 1
    
    if _i = 1
        SetLed(16, 0)
    else
        SetLed(_i-1, 0)
    end
    
    SetLed(_i, 1)
    
    Wait(100)

    _i = (_i + 1)
    
    if (_i>16)
        _i = 1
    end
wend
